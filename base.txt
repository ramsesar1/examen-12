# models.py
from django.db import models

class Usuario(models.Model):
    nombre = models.CharField(max_length=255)
    correo = models.EmailField(unique=True)
    contraseña = models.CharField(max_length=255)
    telefono = models.CharField(max_length=15)
    direccion = models.TextField()

    def __str__(self):
        return self.nombre

# views.py
from django.contrib.auth import authenticate, login
from django.contrib.auth.forms import AuthenticationForm
from django.contrib import messages
from .models import Usuario

def iniciar_sesion(request):
    if request.method == 'POST':
        form = AuthenticationForm(request, request.POST)
        if form.is_valid():
            correo = form.cleaned_data.get('username')  # username es el campo de correo electrónico
            contraseña = form.cleaned_data.get('password')

            if not correo or not contraseña:
                messages.error(request, 'Por favor, ingresa tu correo y contraseña.')
                return render(request, 'myappsitio/login.html', {'form': form})

            # Consultar la base de datos para autenticar al usuario
            usuario = authenticate(request, correo=correo, contraseña=contraseña)

            if usuario is not None:
                login(request, usuario)
                return redirect('nombre_de_tu_vista_inicio')
            else:
                messages.error(request, 'Inicio de sesión fallido. Comprueba tu correo y contraseña.')
                return render(request, 'myappsitio/login.html', {'form': form})
        else:
            messages.error(request, 'El formulario no es válido. Comprueba tus datos.')
            return render(request, 'myappsitio/login.html', {'form': form})
    else:
        form = AuthenticationForm()
    
    return render(request, 'myappsitio/login.html', {'form': form})
